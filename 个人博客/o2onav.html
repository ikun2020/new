CSS部分

   }
   .box2{
    height: 480px;
    float: left;
    margin-left: 10px;
   }
   .img img:last-child{
    float: right
   }
   .img img{
    padding-top: 5px;
   }
   .box1 ul li{
    list-style: none;
    line-height: 40px;
    color: #fff;
   }
   .box1 ul li{
    padding-left: 30px;
   }
   .right{
    width:800px ;

    top: 0px;
    left: 200px;
    display: none;

Javascript部分

<script type="text/javascript">
   $('.box1 li').mouseenter(function(){
    $(this).css({'background':"#eee"});
    $(this).find('span').css({'color':'black'})
    $('.box1 li').not($(this)).css({'background':'#6e6568'});
    $(this).find('.right').show();
    $('.box1 li').not((this).find('.right')).hide();
   })
   $('.box1 li').mouseleave(function(){
    $(this).find('.right').hide();
    $(this).find('span').css({'color':'#fff'});
    $(this).css({'background':'#6e6568'})
   })
 </script>
